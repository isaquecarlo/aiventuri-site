  const camera = new THREE.PerspectiveCamera(45, 1, 0.1, 100);
  camera.position.set(0, 0, 14);

  // Resize
  function resize() {
    const w = canvas.clientWidth, h = canvas.clientHeight;
    renderer.setSize(w, h, false);
    camera.aspect = w / h;
    camera.updateProjectionMatrix();
  }
  resize();
  window.addEventListener('resize', resize);

  // ── LIGHTS ──
  const ambLight = new THREE.AmbientLight(0x1a2550, 1.2);
  scene.add(ambLight);

  const goldLight = new THREE.PointLight(0xc9a84c, 3, 20);
  goldLight.position.set(5, 5, 5);
  scene.add(goldLight);

  const blueLight = new THREE.PointLight(0x2d55b8, 4, 20);
  blueLight.position.set(-5, -3, 4);
  scene.add(blueLight);

  const rimLight = new THREE.DirectionalLight(0x8ab4e8, 1.5);
  rimLight.position.set(0, 10, -5);
  scene.add(rimLight);

  // ── MATERIALS ──
  const sapphireMat = new THREE.MeshPhongMaterial({
    color: 0x1a3580,
    emissive: 0x0a1840,
    specular: 0x4a80ff,
    shininess: 120,
    transparent: true,
    opacity: 0.88,
    side: THREE.DoubleSide,
  });

  const goldMat = new THREE.MeshStandardMaterial({
    color: 0xc9a84c,
    emissive: 0x3a2800,
    metalness: 0.9,
    roughness: 0.15,
  });

  const glowMat = new THREE.MeshBasicMaterial({ color: 0xdfc06a, transparent: true, opacity: 0.8 });

  // ── MAIN GROUP ──
  const mainGroup = new THREE.Group();
  scene.add(mainGroup);

  // ── BUILD LOGO GEOMETRY (V shape + diamond network) ──

  // Vertices of the logo shape (approximated from Venture AI logo)
  // Large V shape
  function makeTriPrism(points2D, depth, mat) {
    const shape = new THREE.Shape();
    shape.moveTo(points2D[0][0], points2D[0][1]);
    for (let i = 1; i < points2D.length; i++) shape.lineTo(points2D[i][0], points2D[i][1]);
    shape.closePath();
    const geo = new THREE.ExtrudeGeometry(shape, { depth, bevelEnabled: true, bevelSize: 0.05, bevelThickness: 0.05, bevelSegments: 2 });
    return new THREE.Mesh(geo, mat);
  }

  // Left crystal shard
  const leftShard = makeTriPrism([
    [-2.2, 0.8],
    [-0.6, -0.3],
    [-0.05, -2.4],
    [-1.0, -2.4],
    [-2.6, -0.4],
  ], 0.5, sapphireMat.clone());
  leftShard.rotation.z = 0.05;
  mainGroup.add(leftShard);

  // Right crystal shard
  const rightShard = makeTriPrism([
    [0.05, -2.4],
    [0.6, -0.3],
    [2.2, 0.8],
    [2.6, -0.4],
    [1.0, -2.4],
  ], 0.5, sapphireMat.clone());
  rightShard.rotation.z = -0.05;
  mainGroup.add(rightShard);

  // Top diamond crystal
  const topDiamond = makeTriPrism([
    [-1.8, 2.8],
    [0, 0.4],
    [1.8, 2.8],
    [0.8, 3.8],
    [-0.8, 3.8],
  ], 0.6, sapphireMat.clone());
  mainGroup.add(topDiamond);

  // ── GOLD NETWORK OVERLAY ──
  const nodePositions = [
    [0, 0, 0.5],         // center
    [-1.6, 1.8, 0.6],   // top-left
    [1.6, 1.8, 0.6],    // top-right
    [0, 3.2, 0.7],       // top
    [-1.2, -1.2, 0.4],  // bottom-left
    [1.2, -1.2, 0.4],   // bottom-right
    [-0.6, 0.8, 0.8],   // mid-left
    [0.6, 0.8, 0.8],    // mid-right
    [0, 1.6, 0.9],       // mid-top
  ];

  const nodeMeshes = [];
  const sphereGeo = new THREE.SphereGeometry(0.1, 10, 10);
  nodePositions.forEach(([x, y, z]) => {
    const s = new THREE.Mesh(sphereGeo, goldMat);
    s.position.set(x, y, z);
    mainGroup.add(s);
    nodeMeshes.push(s);
  });

  // Connections between nodes
  const connections = [
    [0,1],[0,2],[0,6],[0,7],[0,8],
    [1,2],[1,3],[1,6],
    [2,3],[2,7],
    [3,8],[6,7],[6,8],[7,8],
    [0,4],[0,5],[4,5],
    [1,4],[2,5],
  ];

  function makeLine(a, b) {
    const pts = [
      new THREE.Vector3(...nodePositions[a]),
      new THREE.Vector3(...nodePositions[b]),
    ];
    const geo = new THREE.BufferGeometry().setFromPoints(pts);
    const mat = new THREE.LineBasicMaterial({ color: 0xc9a84c, transparent: true, opacity: 0.7 });
    return new THREE.Line(geo, mat);
  }

  connections.forEach(([a, b]) => mainGroup.add(makeLine(a, b)));

  // ── FLOATING CRYSTAL SHARDS (background) ──
  const shards = [];
  for (let i = 0; i < 12; i++) {
    const geo = new THREE.OctahedronGeometry(Math.random() * 0.25 + 0.08, 0);
    const mat = new THREE.MeshPhongMaterial({
      color: i % 2 === 0 ? 0x1e3a8a : 0xc9a84c,
      emissive: i % 2 === 0 ? 0x0a1a50 : 0x2a1800,
      transparent: true,
      opacity: Math.random() * 0.3 + 0.15,
      shininess: 80,
    });
    const mesh = new THREE.Mesh(geo, mat);
    mesh.position.set(
      (Math.random() - 0.5) * 18,
      (Math.random() - 0.5) * 14,
      (Math.random() - 0.5) * 4 - 2
    );
    mesh.userData = {
      vx: (Math.random() - 0.5) * 0.004,
      vy: (Math.random() - 0.5) * 0.004,
      rx: Math.random() * 0.012,
      ry: Math.random() * 0.012,
    };
    scene.add(mesh);
    shards.push(mesh);
  }

  // ── PARTICLE FIELD ──
  const particleCount = 120;
  const pPositions = new Float32Array(particleCount * 3);
  for (let i = 0; i < particleCount; i++) {
    pPositions[i * 3] = (Math.random() - 0.5) * 22;
    pPositions[i * 3 + 1] = (Math.random() - 0.5) * 16;
    pPositions[i * 3 + 2] = (Math.random() - 0.5) * 8 - 3;
  }
  const pGeo = new THREE.BufferGeometry();
  pGeo.setAttribute('position', new THREE.BufferAttribute(pPositions, 3));
  const pMat = new THREE.PointsMaterial({ color: 0xc9a84c, size: 0.04, transparent: true, opacity: 0.4 });
  const particles = new THREE.Points(pGeo, pMat);
  scene.add(particles);

  // ── MOUSE PARALLAX ──
  let mouseX = 0, mouseY = 0, targetX = 0, targetY = 0;
  document.addEventListener('mousemove', (e) => {
    mouseX = (e.clientX / window.innerWidth - 0.5) * 2;
    mouseY = -(e.clientY / window.innerHeight - 0.5) * 2;
  });

  // ── ANIMATION LOOP ──
  let t = 0;
  function animate() {
    requestAnimationFrame(animate);
    t += 0.008;

    // Smooth mouse follow
    targetX += (mouseX - targetX) * 0.04;
    targetY += (mouseY - targetY) * 0.04;

    // Main logo group rotation
    mainGroup.rotation.y = Math.sin(t * 0.4) * 0.25 + targetX * 0.3;
    mainGroup.rotation.x = Math.sin(t * 0.3) * 0.12 + targetY * 0.15;
    mainGroup.rotation.z = Math.sin(t * 0.2) * 0.05;
    mainGroup.position.y = Math.sin(t * 0.5) * 0.18;

    // Gold light orbit
    goldLight.position.x = Math.cos(t * 0.6) * 6;
    goldLight.position.z = Math.sin(t * 0.6) * 6;
    goldLight.position.y = Math.sin(t * 0.4) * 3 + 3;

    // Blue light orbit
    blueLight.position.x = Math.cos(t * 0.5 + Math.PI) * 5;
    blueLight.position.z = Math.sin(t * 0.5 + Math.PI) * 5;

    // Node pulsing
    nodeMeshes.forEach((s, i) => {
      const sc = 1 + Math.sin(t * 1.5 + i * 0.8) * 0.15;
      s.scale.setScalar(sc);
    });

    // Floating shards
    shards.forEach(s => {
      s.position.x += s.userData.vx;
      s.position.y += s.userData.vy;
      if (Math.abs(s.position.x) > 10) s.userData.vx *= -1;
      if (Math.abs(s.position.y) > 8) s.userData.vy *= -1;
      s.rotation.x += s.userData.rx;
      s.rotation.y += s.userData.ry;
    });

    // Particles slow drift
    particles.rotation.y = t * 0.02;

    renderer.render(scene, camera);
  }
  animate();
})();
